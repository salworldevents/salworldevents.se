const header = document.getElementById('header'); let lastY = window.scrollY; window.addEventListener('scroll', () => { const y = window.scrollY; header?.classList.toggle('scrolled', y > 20); if (header) { if (y > lastY && y > 120) header.style.transform = 'translateY(-110%)'; else header.style.transform = 'translateY(0)'; } lastY = y; }); const hamburger = document.getElementById('hamburger'); const nav = document.getElementById('nav'); hamburger?.addEventListener('click', () => nav?.classList.toggle('show')); nav?.querySelectorAll('a').forEach(a => a.addEventListener('click', () => nav.classList.remove('show')));
